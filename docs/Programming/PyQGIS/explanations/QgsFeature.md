# ­ЪДЕ PyQGIS Рђћ QgsFeature
**QgsFeature** рЃгрЃљрЃарЃЏрЃЮрЃљрЃЊрЃњрЃћрЃюрЃА рЃћрЃарЃЌ рЃњрЃћрЃЮрЃњрЃарЃљрЃцрЃўрЃБрЃџ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃА рЃЋрЃћрЃЦрЃбрЃЮрЃарЃБрЃџ рЃцрЃћрЃюрЃљрЃерЃў: рЃЏрЃљрЃА рЃљрЃЦрЃЋрЃА **рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃљ** (QgsGeometry) рЃЊрЃљ **рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃў** (рЃцрЃўрЃџрЃЊрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃАрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў).  
рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА рЃћрЃА рЃљрЃарЃўрЃА РђърЃћрЃарЃЌрЃў рЃарЃўрЃњрЃўРђю рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџ рЃфрЃ«рЃарЃўрЃџрЃерЃў + РђърЃћрЃарЃЌрЃў рЃцрЃўрЃњрЃБрЃарЃљРђю рЃарЃБрЃЎрЃљрЃќрЃћ.

---

## ­ЪЊџ рЃарЃЮрЃЊрЃўрЃА рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ QgsFeature-рЃА?
- рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃЊрЃљрЃАрЃљрЃЏрЃљрЃбрЃћрЃЉрЃџрЃљрЃЊ рЃцрЃћрЃюрЃљрЃерЃў (рЃгрЃћрЃарЃбрЃўрЃџрЃў/рЃ«рЃљрЃќрЃў/рЃърЃЮрЃџрЃўрЃњрЃЮрЃюрЃў)
- рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃљрЃАрЃљрЃЎрЃарЃћрЃцрЃљрЃЊ, рЃгрЃљрЃАрЃљрЃЎрЃўрЃЌрЃ«рЃљрЃЊ рЃЊрЃљ рЃерЃћрЃАрЃљрЃфрЃЋрЃџрЃћрЃџрЃљрЃЊ
- рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃўрЃА рЃЊрЃљрЃАрЃљрЃДрЃћрЃюрЃћрЃЉрЃџрЃљрЃЊ/рЃњрЃљрЃюрЃљрЃ«рЃџрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА
- рЃЉрЃљрЃцрЃћрЃарЃћрЃЉрЃўрЃА/рЃњрЃљрЃЊрЃљрЃцрЃљрЃарЃЋрЃћрЃЉрЃўрЃА/рЃАрЃ«рЃЋрЃљ рЃљрЃюрЃљрЃџрЃўрЃќрЃўрЃА рЃерЃћрЃЊрЃћрЃњрЃћрЃЉрЃўрЃА рЃЕрЃљрЃАрЃљрЃгрЃћрЃарЃљрЃЊ

---

## ­ЪДа рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃЎрЃЮрЃюрЃфрЃћрЃцрЃфрЃўрЃћрЃЉрЃў
- **Feature (рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў)** Рђћ рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃўрЃАрЃљ рЃЊрЃљ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃўрЃА рЃћрЃарЃЌрЃЮрЃЉрЃџрЃўрЃЮрЃЉрЃљ
- **FID/ID** Рђћ рЃцрЃћрЃюрЃљрЃерЃў рЃБрЃюрЃўрЃЎрЃљрЃџрЃБрЃарЃў рЃўрЃЊрЃћрЃюрЃбрЃўрЃцрЃўрЃЎрЃљрЃбрЃЮрЃарЃў (рЃњрЃљрЃюрЃАрЃ«рЃЋрЃљрЃЋрЃЊрЃћрЃЉрЃљ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃўрЃА рЃАрЃЋрЃћрЃбрЃерЃў рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў РђъIDРђю-рЃАрЃњрЃљрЃю)
- **Geometry** Рђћ QgsGeometry (Point/LineString/Polygon/MultipartsРђд)
- **Attributes** Рђћ рЃАрЃўрЃљ/рЃџрЃўрЃАрЃбрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃўрЃАрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ рЃцрЃћрЃюрЃўрЃА `QgsFields()` рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА

---

## РюЁ рЃАрЃгрЃарЃљрЃцрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў Рђћ QgsFeature рЃерЃћрЃЦрЃЏрЃюрЃљ рЃЊрЃљ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃўрЃА рЃЊрЃљрЃДрЃћрЃюрЃћрЃЉрЃљ
```python
from qgis.core import QgsFeature, QgsGeometry, QgsPointXY

feat = QgsFeature()                              # рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў
geom = QgsGeometry.fromPointXY(QgsPointXY(44.8, 41.7))
feat.setGeometry(geom)                           # рЃЋрЃљрЃюрЃўрЃГрЃћрЃЉрЃЌ рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃљрЃА
feat.setAttributes([1, "Tbilisi", 1110000])     # рЃЋрЃљрЃюрЃўрЃГрЃћрЃЉрЃЌ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃА (рЃБрЃЏрЃ»рЃЮрЃЉрЃћрЃАрЃўрЃљ рЃцрЃћрЃюрЃљрЃАрЃЌрЃљрЃю рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃАрЃў рЃарЃўрЃњрЃўрЃЌрЃЮрЃЉрЃўрЃЌ)
```

> рЃерЃћрЃюрЃўрЃерЃЋрЃюрЃљ: `setAttributes()` рЃАрЃўрЃљрЃерЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃЌрЃљ рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљ рЃБрЃюрЃЊрЃљ рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЮрЃЊрЃћрЃА рЃцрЃћрЃюрЃўрЃА `QgsFields()` рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА (рЃљрЃю рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћ `setAttribute(index, value)`).

## ­ЪДф рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃцрЃћрЃюрЃљрЃАрЃЌрЃљрЃю Рђћ рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃЕрЃљрЃгрЃћрЃарЃљ
```python
from qgis.core import QgsProject, QgsVectorLayer, QgsField, QgsFields, QgsWkbTypes, QgsFeature, QgsGeometry, QgsPointXY
from PyQt5.QtCore import QVariant

# рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃЏрЃћрЃ«рЃАрЃўрЃћрЃарЃћрЃЉрЃўрЃА рЃцрЃћрЃюрЃљ (Point, EPSG:4326)
vl = QgsVectorLayer("Point?crs=EPSG:4326", "Cities", "memory")
pr = vl.dataProvider()

# рЃЊрЃљрЃЋрЃљрЃЏрЃљрЃбрЃЮрЃЌ рЃЋрЃћрЃџрЃћрЃЉрЃў
pr.addAttributes([QgsField("ID", QVariant.Int), QgsField("Name", QVariant.String), QgsField("Pop", QVariant.Int)])
vl.updateFields()

# рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў
f = QgsFeature(vl.fields())                      # рЃњрЃљрЃЊрЃљрЃЋрЃфрЃћрЃЏрЃЌ рЃЋрЃћрЃџрЃћрЃЉрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА
f.setGeometry(QgsGeometry.fromPointXY(QgsPointXY(44.8, 41.7)))
f["ID"] = 1                                      # рЃљрЃџрЃбрЃћрЃарЃюрЃљрЃбрЃўрЃЋрЃљ: f.setAttribute("ID", 1)
f["Name"] = "Tbilisi"
f["Pop"] = 1110000

pr.addFeatures([f])                              # рЃЕрЃљрЃЋрЃљрЃЏрЃљрЃбрЃЮрЃЌ рЃцрЃћрЃюрЃљрЃерЃў
vl.updateExtents()

QgsProject.instance().addMapLayer(vl)
```

## ­ЪћЂ рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃгрЃљрЃЎрЃўрЃЌрЃ«рЃЋрЃљ рЃЊрЃљ рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃљ (рЃарЃћрЃЊрЃљрЃЦрЃбрЃўрЃарЃћрЃЉрЃўрЃА рЃарЃћрЃЪрЃўрЃЏрЃў)
```python
# рЃарЃћрЃЊрЃљрЃЦрЃбрЃўрЃарЃћрЃЉрЃўрЃА рЃарЃћрЃЪрЃўрЃЏрЃў рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃўрЃА/рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃўрЃА рЃерЃћрЃАрЃљрЃфрЃЋрЃџрЃћрЃџрЃљрЃЊ
vl.startEditing()

for f in vl.getFeatures():                       # рЃўрЃбрЃћрЃарЃљрЃфрЃўрЃљ рЃДрЃЋрЃћрЃџрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃќрЃћ
    if f["Name"] == "Tbilisi":
        vl.changeAttributeValue(f.id(), vl.fields().indexOf("Pop"), 1200000)  # Pop = 1.2m

# рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ (рЃЏрЃљрЃњ., рЃЏрЃўрЃџрЃўрЃЏрЃћрЃбрЃарЃБрЃџрЃў рЃерЃўрЃцрЃбрЃў)
for f in vl.getFeatures():
    g = f.geometry()
    if g and g.isMultipart() is False:
        p = g.asPoint()
        new_g = QgsGeometry.fromPointXY(QgsPointXY(p.x() + 0.001, p.y()))
        vl.changeGeometry(f.id(), new_g)

vl.commitChanges()                               # рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃўрЃА рЃерЃћрЃюрЃљрЃ«рЃЋрЃљ
```

> TIP: `vl.rollBack()` рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃўрЃАрЃљрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃўрЃА рЃњрЃљрЃАрЃљрЃБрЃЦрЃЏрЃћрЃЉрЃџрЃљрЃЊ.

---

## ­ЪД« рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮ рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў
- `QgsFeature(fields=None)` Рђћ рЃЎрЃЮрЃюрЃАрЃбрЃарЃБрЃЦрЃбрЃЮрЃарЃў; рЃАрЃБрЃарЃЋрЃўрЃџрЃўрЃАрЃљрЃЏрЃћрЃЉрЃа рЃЋрЃћрЃџрЃћрЃЉрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃЌ
- `setGeometry(QgsGeometry)` / `geometry()`
- `setAttributes(list)` / `attributes()`
- `setAttribute(index_or_name, value)` / `attribute(index_or_name)`
- `id()` Рђћ рЃцрЃћрЃюрЃўрЃА рЃерЃўрЃЊрЃљ рЃўрЃЊрЃћрЃюрЃбрЃўрЃцрЃўрЃЎрЃљрЃбрЃЮрЃарЃў (FID)
- `isValid()` Рђћ рЃљрЃЦрЃЋрЃА рЃЌрЃБ рЃљрЃарЃљ рЃЏрЃўрЃюрЃўрЃГрЃћрЃЉрЃБрЃџрЃў рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃљ/рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃў рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃАрЃЮрЃЉрЃљрЃерЃў

---

## ­ЪДи рЃЏрЃарЃљрЃЋрЃљрЃџрЃюрЃљрЃгрЃўрЃџрЃўрЃљрЃюрЃў рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃљ (Multipart)
```python
from qgis.core import QgsGeometry, QgsPointXY

# рЃЮрЃарЃў рЃгрЃћрЃарЃбрЃўрЃџрЃўрЃАрЃњрЃљрЃю Multipart Point
mp = QgsGeometry.fromMultiPointXY([QgsPointXY(44.8, 41.7), QgsPointXY(41.6, 41.65)])
f = QgsFeature(vl.fields())
f.setGeometry(mp)
f.setAttributes([2, "Cities MP", None])
vl.dataProvider().addFeatures([f])
vl.updateExtents()
```

> рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћ `geometry().isMultipart()` рЃарЃљрЃЌрЃљ рЃњрЃљрЃўрЃњрЃЮ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃЏрЃарЃљрЃЋрЃљрЃџрЃюрЃљрЃгрЃўрЃџрЃўрЃљрЃюрЃў.

---

## ­ЪД░ QgsFeature рЃерЃарЃћрЃћрЃЉрЃўрЃЊрЃљрЃю рЃљрЃЏрЃЮрЃдрЃћрЃЉрЃљ (Selection / Filter)
```python
# 1) Feature iterator + expression filter
expr = '"Pop" > 1000000'                         # QGIS expression
for f in vl.getFeatures(expr):
    print(f["Name"], f["Pop"])

# 2) рЃЏрЃљрЃўрЃЊрЃћрЃюрЃбрЃўрЃцрЃўрЃфрЃўрЃарЃћрЃЉрЃћрЃџрЃў Feature ID-рЃћрЃЉрЃўрЃЌ
ids = [f.id() for f in vl.getFeatures() if f["Name"].startswith("T")]
for fid in ids:
    feat = next(vl.getFeatures(f"id = {fid}"))   # рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў FID-рЃўрЃА рЃЏрЃЮрЃФрЃћрЃЉрЃюрЃљ
    print("Selected:", feat["Name"])
```

## ­ЪД▒ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃў рЃЊрЃљ рЃњрЃљрЃЋрЃарЃфрЃћрЃџрЃћрЃЉрЃБрЃџрЃў рЃ«рЃљрЃцрЃљрЃюрЃњрЃћрЃЉрЃў
- **рЃЋрЃћрЃџрЃћрЃЉрЃўрЃА рЃЉрЃЏрЃўрЃА рЃЊрЃљрЃарЃдрЃЋрЃћрЃЋрЃљ** Рђћ `setAttributes()` рЃАрЃўрЃўрЃА рЃАрЃўрЃњрЃарЃФрЃћ рЃљрЃа рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ `QgsFields()` рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА Рєњ рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћ `setAttribute()` рЃўрЃюрЃЊрЃўрЃЋрЃўрЃЊрЃБрЃљрЃџрЃБрЃарЃљрЃЊ.
- **рЃћрЃЊрЃўрЃб рЃарЃћрЃЪрЃўрЃЏрЃўрЃА рЃЊрЃљрЃЋрЃўрЃгрЃДрЃћрЃЉрЃљ** Рђћ `changeAttributeValue` / `changeGeometry` рЃЏрЃЮрЃўрЃЌрЃ«рЃЮрЃЋрЃА `startEditing()` рЃЊрЃљ `commitChanges()`.
- **FID vs Attribute ID** Рђћ `f.id()` != `f["ID"]` (рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃерЃўрЃЊрЃљ рЃбрЃћрЃЦрЃюрЃўрЃЎрЃБрЃарЃў рЃўрЃЊрЃћрЃюрЃбрЃўрЃцрЃўрЃЎрЃљрЃбрЃЮрЃарЃўрЃљ).  
- **CRS/рЃћрЃарЃЌрЃћрЃБрЃџрЃћрЃЉрЃў** Рђћ `geometry().area()/length()` рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃерЃћрЃЊрЃћрЃњрЃА рЃцрЃћрЃюрЃўрЃА CRS-рЃўрЃА рЃћрЃарЃЌрЃћрЃБрЃџрЃћрЃЉрЃерЃў.

## ­ЪДГ рЃЏрЃфрЃўрЃарЃћ рЃљрЃЏрЃЮрЃфрЃљрЃюрЃћрЃЉрЃў рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА
1) рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃЏрЃћрЃ«рЃАрЃўрЃћрЃарЃћрЃЉрЃўрЃА рЃцрЃћрЃюрЃљ Point рЃбрЃўрЃърЃўрЃЌ, рЃЋрЃћрЃџрЃћрЃЉрЃўрЃЌ: `ID(Int), Name(String), Elev(Double)` рЃЊрЃљ рЃЊрЃљрЃљрЃЏрЃљрЃбрЃћ 5 рЃгрЃћрЃарЃбрЃўрЃџрЃў.  
2) рЃўрЃърЃЮрЃЋрЃћ рЃДрЃЋрЃћрЃџрЃљ рЃгрЃћрЃарЃбрЃўрЃџрЃў, рЃАрЃљрЃЊрЃљрЃф `Elev > 1200`, рЃЊрЃљ рЃЊрЃљрЃљрЃЏрЃљрЃбрЃћ рЃЏрЃљрЃЌ рЃљрЃ«рЃљрЃџ рЃЋрЃћрЃџрЃерЃў `High=True`.  
3) рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃ«рЃљрЃќрЃЮрЃЋрЃљрЃюрЃў рЃцрЃћрЃюрЃўрЃЊрЃљрЃю рЃљрЃўрЃдрЃћ рЃЌрЃўрЃЌрЃЮ feature рЃЊрЃљ рЃЕрЃљрЃгрЃћрЃарЃћ рЃЏрЃўрЃАрЃў рЃАрЃўрЃњрЃарЃФрЃћ рЃЋрЃћрЃџрЃерЃў `Len_m` (рЃњрЃљрЃўрЃЌрЃЋрЃљрЃџрЃўрЃАрЃгрЃўрЃюрЃћ CRS рЃЏрЃћрЃбрЃарЃћрЃЉрЃерЃў).  
4) рЃерЃћрЃЦрЃЏрЃћрЃюрЃў Multipart Polygon рЃћрЃарЃЌрЃў Feature-рЃўрЃЌ рЃЮрЃарЃў рЃАрЃљрЃЏрЃЎрЃБрЃЌрЃ«рЃћрЃЊрЃўрЃЌ рЃЊрЃљ рЃЕрЃљрЃљрЃЏрЃљрЃбрЃћ рЃцрЃћрЃюрЃљрЃерЃў.

---

## ­Ъћџ рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ
- **QgsFeature** рЃљрЃарЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃАрЃљрЃЏрЃерЃћрЃюрЃћрЃЉрЃџрЃЮ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃў рЃЋрЃћрЃЦрЃбрЃЮрЃарЃБрЃџ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃерЃў.  
- рЃЏрЃљрЃАрЃЌрЃљрЃю рЃћрЃарЃЌрЃљрЃЊ рЃўрЃДрЃћрЃюрЃћрЃЉ **QgsGeometry**-рЃАрЃљ рЃЊрЃљ **QgsFields/QgsField**-рЃА.  
- рЃАрЃгрЃЮрЃарЃў рЃћрЃЊрЃўрЃб-рЃфрЃўрЃЎрЃџрЃў: `startEditing()` Рєњ рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃў Рєњ `commitChanges()` (рЃљрЃю `rollBack()`).  
- рЃўрЃЏрЃБрЃерЃљрЃЋрЃћ рЃњрЃљрЃЏрЃГрЃЋрЃўрЃарЃЋрЃљрЃџрЃћ рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃћрЃЉрЃўрЃА рЃАрЃЦрЃћрЃЏрЃўрЃЌ рЃЊрЃљ CRS-рЃћрЃЉрЃўрЃА рЃфрЃЮрЃЊрЃюрЃўрЃЌ.
