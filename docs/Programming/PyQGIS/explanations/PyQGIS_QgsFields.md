# ­ЪЊЮ PyQGIS: **QgsFields()** рЃарЃљ рЃљрЃарЃўрЃА рЃЊрЃљ рЃарЃљрЃА рЃљрЃЎрЃћрЃЌрЃћрЃЉрЃА?

## РќХ№ИЈ рЃарЃљ рЃљрЃарЃўрЃА QgsFields()?  
`QgsFields` рЃљрЃарЃўрЃА **рЃЎрЃџрЃљрЃАрЃў PyQGIS-рЃерЃў**, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃгрЃљрЃарЃЏрЃЮрЃљрЃЊрЃњрЃћрЃюрЃА **рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџрЃў рЃфрЃ«рЃарЃўрЃџрЃўрЃА рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃџрЃћрЃЦрЃфрЃўрЃљрЃА**.  
рЃљрЃюрЃБ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃљрЃ«рЃљрЃџ рЃЋрЃћрЃЦрЃбрЃЮрЃарЃБрЃџ рЃерЃарЃћрЃА (рЃЏрЃљрЃњ., shapefile), рЃЏрЃўрЃА рЃфрЃ«рЃарЃўрЃџрЃерЃў рЃљрЃБрЃфрЃўрЃџрЃћрЃЉрЃџрЃљрЃЊ рЃБрЃюрЃЊрЃљ рЃњрЃљрЃюрЃЋрЃАрЃљрЃќрЃдрЃЋрЃарЃЮрЃЌ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃў (ID, рЃАрЃљрЃ«рЃћрЃџрЃў, рЃљрЃАрЃљрЃЎрЃў рЃЊрЃљ рЃљ.рЃе.). рЃАрЃгрЃЮрЃарЃћрЃЊ рЃћрЃА рЃАрЃЋрЃћрЃбрЃћрЃЉрЃў рЃўрЃюрЃљрЃ«рЃћрЃЉрЃљ **QgsFields()** рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃерЃў.
рЃЋрЃћрЃџрЃћрЃЉрЃў (Fields) рЃљрЃарЃўрЃА рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў, рЃарЃЮрЃЏрЃџрЃћрЃЉрЃўрЃф рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃћрЃю рЃцрЃћрЃюрЃљрЃА (Layer) рЃљрЃю рЃцрЃљрЃўрЃџрЃерЃў рЃерЃћрЃюрЃљрЃ«рЃБрЃџ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃА. рЃЌрЃўрЃЌрЃЮрЃћрЃБрЃџрЃў рЃЋрЃћрЃџрЃў рЃљрЃЦрЃЋрЃА рЃАрЃљрЃ«рЃћрЃџрЃў, рЃбрЃўрЃърЃў (рЃЏрЃљрЃњ., рЃбрЃћрЃЦрЃАрЃбрЃў, рЃарЃўрЃфрЃ«рЃЋрЃў) рЃЊрЃљ рЃАрЃ«рЃЋрЃљ рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃћрЃЉрЃў.
layerfield = QgsFields()
рЃљрЃЏ рЃЎрЃЮрЃЊрЃўрЃА рЃерЃћрЃЊрЃћрЃњрЃљрЃЊ:

рЃўрЃЦрЃЏрЃюрЃћрЃЉрЃљ рЃфрЃљрЃарЃўрЃћрЃџрЃў QgsFields рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў.
рЃћрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃљрЃ«рЃљрЃџрЃў рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџрЃў рЃЋрЃћрЃџрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА рЃљрЃю рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃЋрЃћрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃюрЃљрЃ«рЃљрЃЊ рЃцрЃћрЃюрЃўрЃАрЃЌрЃЋрЃўрЃА.

---

## ­ЪЊд рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ
- рЃўрЃюрЃљрЃ«рЃљрЃЋрЃА **QgsField** рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃА (рЃљрЃюрЃБ рЃЌрЃўрЃЌрЃЮрЃћрЃБрЃџрЃў рЃАрЃЋрЃћрЃбрЃўрЃА рЃљрЃдрЃгрЃћрЃарЃљрЃА).  
- рЃўрЃФрЃџрЃћрЃЋрЃљ рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃЏрЃўрЃЋрЃБрЃЏрЃљрЃбрЃЮрЃЌ, рЃгрЃљрЃЋрЃерЃљрЃџрЃЮрЃЌ рЃљрЃю рЃЏрЃЮрЃЋрЃўрЃФрЃўрЃЮрЃЌ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃў.  
- рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃљрЃ«рЃљрЃџ рЃерЃарЃћрЃерЃў рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃЕрЃљрЃАрЃљрЃгрЃћрЃарЃљрЃЊ.

---

## ­ЪД▒ рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў РђЊ рЃљрЃ«рЃљрЃџрЃў рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџрЃў рЃфрЃ«рЃарЃўрЃџрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

```python
from qgis.core import QgsFields, QgsField
from PyQt5.QtCore import QVariant

# рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃфрЃљрЃарЃўрЃћрЃџ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃџрЃћрЃЦрЃфрЃўрЃљрЃА
fields = QgsFields()

# рЃЋрЃљрЃЏрЃљрЃбрЃћрЃЉрЃЌ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃА
fields.append(QgsField("ID", QVariant.Int))
fields.append(QgsField("Name", QVariant.String))
fields.append(QgsField("Age", QVariant.Double))

# рЃљрЃ«рЃџрЃљ fields рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃўрЃюрЃљрЃ«рЃљрЃЋрЃА 3 рЃАрЃЋрЃћрЃбрЃА: ID, Name, Age
print(fields.count())   # рЃњрЃљрЃЏрЃЮрЃўрЃбрЃљрЃюрЃА 3
```

---

## ­ЪЊі рЃарЃЮрЃњрЃЮрЃа рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА?

- `QgsField` РђЊ рЃљрЃдрЃгрЃћрЃарЃА рЃћрЃарЃЌ рЃАрЃЋрЃћрЃбрЃА (рЃАрЃљрЃ«рЃћрЃџрЃў + рЃбрЃўрЃърЃў).  
- `QgsFields` РђЊ рЃљрЃарЃўрЃА **рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃА рЃ»рЃњрЃБрЃцрЃў**.  
- рЃарЃЮрЃфрЃљ `QgsVectorFileWriter`РђЊрЃА рЃЋрЃљрЃгрЃЋрЃЊрЃўрЃЌ `QgsFields`, рЃўрЃА рЃЦрЃЏрЃюрЃўрЃА рЃљрЃ«рЃљрЃџ shapefile-рЃА рЃЏрЃўрЃЌрЃўрЃЌрЃћрЃЉрЃБрЃџрЃў рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃЌ.

---

## ­Ъћј рЃарЃљрЃбрЃЮрЃЏ рЃљрЃарЃўрЃА рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў?

1. **рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљ** РђЊ рЃЌрЃБ рЃљрЃа рЃњрЃЋрЃљрЃЦрЃЋрЃА рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџрЃў рЃАрЃЋрЃћрЃбрЃћрЃЉрЃў, рЃЋрЃћрЃа рЃерЃћрЃЋрЃўрЃюрЃљрЃ«рЃљрЃЋрЃЌ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃА.  
2. **рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ** РђЊ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃБрЃџрЃљрЃЊ рЃЋрЃљрЃЎрЃЮрЃюрЃбрЃарЃЮрЃџрЃћрЃЉрЃЌ рЃарЃљ рЃбрЃўрЃърЃўрЃА рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃљ рЃЕрЃљрЃўрЃгрЃћрЃарЃћрЃЉрЃљ (рЃЏрЃЌрЃћрЃџрЃў рЃарЃўрЃфрЃ«рЃЋрЃў, рЃбрЃћрЃЦрЃАрЃбрЃў, рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊрЃў рЃЊрЃљ рЃљ.рЃе.).  
3. **рЃљрЃЋрЃбрЃЮрЃЏрЃљрЃбрЃўрЃќрЃљрЃфрЃўрЃљ** РђЊ рЃарЃЮрЃфрЃљ рЃЉрЃћрЃЋрЃарЃў shapefile рЃБрЃюрЃЊрЃљ рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃЌ, `QgsFields()` рЃљрЃЏ рЃърЃарЃЮрЃфрЃћрЃАрЃА рЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃА.

---

## ­Ъџђ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў: рЃерЃарЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃЌ
```python
from qgis.core import QgsVectorFileWriter, QgsCoordinateReferenceSystem, QgsWkbTypes, QgsFeature, QgsGeometry, QgsPointXY

# рЃцрЃљрЃўрЃџрЃўрЃА рЃЏрЃўрЃАрЃљрЃЏрЃљрЃарЃЌрЃў
fn = r"C:\Users\Public\Documents\GIS\new_points.shp"

# рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃА
fields = QgsFields()
fields.append(QgsField("ID", QVariant.Int))
fields.append(QgsField("Name", QVariant.String))

# рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃерЃарЃћрЃА
writer = QgsVectorFileWriter(fn, "UTF-8", fields, QgsWkbTypes.Point,
                             QgsCoordinateReferenceSystem("EPSG:32638"), "ESRI Shapefile")

# рЃЋрЃљрЃЏрЃљрЃбрЃћрЃЉрЃЌ рЃгрЃћрЃарЃбрЃўрЃџрЃА
feat = QgsFeature()
feat.setGeometry(QgsGeometry.fromPointXY(QgsPointXY(356000, 4679000)))
feat.setAttributes([1, "Lake"])
writer.addFeature(feat)

del(writer)  # рЃЋрЃ«рЃБрЃарЃљрЃЋрЃЌ writer-рЃА

iface.addVectorLayer(fn, "", "ogr")  # рЃЋрЃљрЃЏрЃљрЃбрЃћрЃЉрЃЌ QGIS-рЃерЃў
```

## ­Ъј» рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ
- **`QgsField`** РђЊ рЃћрЃарЃЌ рЃАрЃЋрЃћрЃбрЃА рЃљрЃдрЃгрЃћрЃарЃА.  
- **`QgsFields`** РђЊ рЃАрЃЋрЃћрЃбрЃћрЃЉрЃўрЃА рЃАрЃарЃБрЃџрЃў рЃАрЃўрЃљрЃљ (рЃљрЃбрЃарЃўрЃЉрЃБрЃбрЃБрЃџрЃў рЃфрЃ«рЃарЃўрЃџрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ).  
- рЃЌрЃБ рЃњрЃўрЃюрЃЊрЃљ QGIS-рЃерЃў рЃљрЃ«рЃљрЃџрЃў рЃцрЃљрЃўрЃџрЃў рЃерЃћрЃЦрЃЏрЃюрЃљ, `QgsFields()` рЃљрЃБрЃфрЃўрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ.
