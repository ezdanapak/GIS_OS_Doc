# ­ЪЊў рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў Python-рЃерЃў  

---

## ­Ъћ╣ рЃарЃљ рЃљрЃарЃўрЃА рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў?

Python-рЃА рЃљрЃЦрЃЋрЃА рЃЏрЃарЃљрЃЋрЃљрЃџрЃў **рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў (built-in)** рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃДрЃЮрЃЋрЃћрЃџрЃЊрЃдрЃўрЃБрЃа рЃљрЃЏрЃЮрЃфрЃљрЃюрЃћрЃЉрЃерЃў  
рЃЊрЃљ рЃљрЃа рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃўрЃЌрЃў рЃЏрЃЮрЃЊрЃБрЃџрЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљрЃА.

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ:
```python
x = -10
print(abs(x))  # рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃљрЃЉрЃАрЃЮрЃџрЃБрЃбрЃБрЃа рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА Рєњ 10

y = 5.8
print(round(y))  # рЃљрЃЏрЃарЃњрЃЋрЃљрЃџрЃћрЃЉрЃА Рєњ 6
­Ъћ╣ рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃЌрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў
рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ	рЃљрЃдрЃгрЃћрЃарЃљ	рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў	рЃерЃћрЃЊрЃћрЃњрЃў
abs(x)	рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃљрЃЉрЃАрЃЮрЃџрЃБрЃбрЃБрЃа рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА	abs(-7)	7
pow(x, y)	рЃљрЃДрЃљрЃЋрЃА x рЃарЃўрЃфрЃ«рЃЋрЃў y рЃ«рЃљрЃарЃўрЃАрЃ«рЃерЃў	pow(2, 3)	8
round(x, n)	рЃљрЃЏрЃарЃњрЃЋрЃљрЃџрЃћрЃЉрЃА рЃарЃўрЃфрЃ«рЃЋрЃА n рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊрЃўрЃА рЃАрЃўрЃќрЃБрЃАрЃбрЃўрЃЌ	round(3.14159, 2)	3.14
max(a, b, c)	рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃЏрЃљрЃЦрЃАрЃўрЃЏрЃљрЃџрЃБрЃа рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА	max(5, 8, 2)	8
min(a, b, c)	рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃЏрЃўрЃюрЃўрЃЏрЃљрЃџрЃБрЃа рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА	min(5, 8, 2)	2
sum(list)	рЃ»рЃљрЃЏрЃЮрЃЉрЃА рЃАрЃўрЃљрЃерЃў рЃљрЃарЃАрЃћрЃЉрЃБрЃџ рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃА	sum([1, 2, 3])	6

```
## рЃарЃўрЃфрЃ«рЃЋрЃўрЃЌ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃЌрЃљрЃю рЃАрЃљрЃЏрЃБрЃерЃљрЃЮ рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў, Math рЃЏрЃЮрЃЊрЃБрЃџрЃў рЃЊрЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў

­ЪД« Math рЃЏрЃЮрЃЊрЃБрЃџрЃў
рЃЌрЃБ рЃњрЃЋрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃБрЃцрЃарЃЮ рЃарЃЌрЃБрЃџрЃў рЃЏрЃљрЃЌрЃћрЃЏрЃљрЃбрЃўрЃЎрЃБрЃарЃў рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃћрЃЉрЃў Рђћ рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ math рЃЏрЃЮрЃЊрЃБрЃџрЃА.
рЃћрЃА рЃЏрЃЮрЃЊрЃБрЃџрЃў рЃЏрЃЮрЃўрЃфрЃљрЃЋрЃА рЃЏрЃарЃљрЃЋрЃљрЃџ рЃцрЃБрЃюрЃЦрЃфрЃўрЃљрЃА: Рѕџ, sin, cos, log, ¤ђ рЃЊрЃљ рЃљ.рЃе.

```python
Copy code
import math

print(math.sqrt(16))    # рЃЎрЃЋрЃљрЃЊрЃарЃљрЃбрЃБрЃџрЃў рЃцрЃћрЃАрЃЋрЃў Рєњ 4.0
print(math.pi)          # рЃърЃў Рєњ 3.141592653589793
print(math.cos(0))      # cos(0) Рєњ 1.0
print(math.log(10))     # рЃЉрЃБрЃюрЃћрЃЉрЃарЃўрЃЋрЃў рЃџрЃЮрЃњрЃљрЃарЃўрЃЌрЃЏрЃў Рєњ 2.302585...

```

## ­ЪЊд Math рЃЏрЃЮрЃЊрЃБрЃџрЃўрЃА рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў

| рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ | рЃљрЃдрЃгрЃћрЃарЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў | рЃерЃћрЃЊрЃћрЃњрЃў |
|----------|----------|-----------|---------|
| `math.sqrt(x)` | рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃцрЃћрЃАрЃЋрЃА | `import math`<br>`print(math.sqrt(25))` | `5.0` |
| `math.pow(x, y)` | рЃљрЃДрЃљрЃЋрЃА `x`-рЃА `y` рЃ«рЃљрЃарЃўрЃАрЃ«рЃерЃў | `import math`<br>`print(math.pow(2, 4))` | `16.0` |
| `math.floor(x)` | рЃљрЃЏрЃарЃњрЃЋрЃљрЃџрЃћрЃЉрЃА рЃЦрЃЋрЃћрЃЏрЃЮрЃЌ (рЃЏрЃЌрЃћрЃџрЃљрЃЏрЃЊрЃћ) | `import math`<br>`print(math.floor(3.9))` | `3` |
| `math.ceil(x)` | рЃљрЃЏрЃарЃњрЃЋрЃљрЃџрЃћрЃЉрЃА рЃќрЃћрЃЏрЃЮрЃЌ (рЃЏрЃЌрЃћрЃџрЃљрЃЏрЃЊрЃћ) | `import math`<br>`print(math.ceil(3.1))` | `4` |
| `math.pi` | рЃЏрЃБрЃЊрЃЏрЃўрЃЋрЃљ ¤ђ | `import math`<br>`print(math.pi)` | `3.1415...` |
| `math.e` | рЃЏрЃБрЃЊрЃЏрЃўрЃЋрЃљ e | `import math`<br>`print(math.e)` | `2.7182...` |




­Ъј▓ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў (Random рЃЏрЃЮрЃЊрЃБрЃџрЃў)
рЃЌрЃБ рЃњрЃЋрЃўрЃюрЃЊрЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃўрЃА рЃњрЃћрЃюрЃћрЃарЃљрЃфрЃўрЃљ, рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ random рЃЏрЃЮрЃЊрЃБрЃџрЃА.

```python
import random

print(random.random())       # 0 рЃЊрЃљ 1 рЃерЃЮрЃарЃўрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊрЃў
print(random.randint(1, 10)) # рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃЏрЃЌрЃћрЃџрЃў рЃарЃўрЃфрЃ«рЃЋрЃў 1-рЃЊрЃљрЃю 10-рЃЏрЃЊрЃћ
print(random.choice([2, 4, 6, 8])) # рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃљрЃарЃЕрЃћрЃЋрЃљрЃюрЃў рЃАрЃўрЃўрЃЊрЃљрЃю

```
­ЪЊд рЃЏрЃЮрЃЎрЃџрЃћ рЃфрЃ«рЃарЃўрЃџрЃў: Random рЃЏрЃЮрЃЊрЃБрЃџрЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў

| рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ | рЃљрЃдрЃгрЃћрЃарЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў | рЃерЃћрЃЊрЃћрЃњрЃў (рЃЏрЃўрЃљрЃ«рЃџрЃЮрЃћрЃЉрЃўрЃЌрЃў) |
|----------|----------|-----------|------------------------|
| `random.random()` | рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌ рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊ рЃарЃўрЃфрЃ«рЃЋрЃА 0-рЃЊрЃљрЃю 1-рЃЏрЃЊрЃћ. | ```import random print(random.random())``` | 0.7345 |
| `random.randint(a, b)` | рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌ рЃЏрЃЌрЃћрЃџ рЃарЃўрЃфрЃ«рЃЋрЃА `[a, b]` рЃЊрЃўрЃљрЃърЃљрЃќрЃЮрЃюрЃерЃў (рЃЕрЃљрЃЌрЃЋрЃџрЃўрЃЌ `b`). | ```import random print(random.randint(5, 15))``` | 11 |
| `random.choice(list)` | рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃА рЃАрЃўрЃўрЃЊрЃљрЃю. | ```import random print(random.choice(['a', 'b', 'c']))``` | `'b'` |
| `random.uniform(a, b)` | рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌ рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊ рЃарЃўрЃфрЃ«рЃЋрЃА `[a, b]` рЃЊрЃўрЃљрЃърЃљрЃќрЃЮрЃюрЃерЃў. | ```import random print(random.uniform(1, 5))``` | 3.64 |
| `random.shuffle(list)` | рЃљрЃџрЃљрЃњрЃћрЃЉрЃА рЃАрЃўрЃўрЃА рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў рЃгрЃћрЃАрЃўрЃЌ (рЃфрЃЋрЃџрЃўрЃА рЃЌрЃљрЃЋрЃЊрЃљрЃърЃўрЃарЃЋрЃћрЃџ рЃАрЃўрЃљрЃА). | ```import random nums = [1, 2, 3, 4] random.shuffle(nums) print(nums)``` | `[3, 1, 4, 2]` *(рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌрЃў)* |


­ЪДЕ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃљрЃЏрЃЮрЃфрЃљрЃюрЃћрЃЉрЃў
­ЪДа рЃЏрЃљрЃарЃбрЃўрЃЋрЃў
рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃА рЃерЃћрЃћрЃЎрЃўрЃЌрЃ«рЃћ рЃарЃўрЃфрЃ«рЃЋрЃў рЃЊрЃљ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃЏрЃўрЃАрЃў рЃљрЃЉрЃАрЃЮрЃџрЃБрЃбрЃБрЃарЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ.

рЃўрЃърЃЮрЃЋрЃћ рЃАрЃљрЃЏ рЃарЃўрЃфрЃ«рЃЋрЃА рЃерЃЮрЃарЃўрЃА рЃЏрЃљрЃЦрЃАрЃўрЃЏрЃљрЃџрЃБрЃарЃў рЃЊрЃљ рЃЏрЃўрЃюрЃўрЃЏрЃљрЃџрЃБрЃарЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў.

рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃЏрЃўрЃћрЃа рЃерЃћрЃДрЃЋрЃљрЃюрЃўрЃџрЃў рЃарЃўрЃфрЃ«рЃЋрЃў рЃњрЃљрЃљрЃЏрЃарЃњрЃЋрЃљрЃџрЃћ рЃЮрЃарЃў рЃљрЃЌрЃгрЃўрЃџрЃљрЃЊрЃў рЃАрЃўрЃќрЃБрЃАрЃбрЃўрЃЌ.

РџЎ№ИЈ рЃАрЃљрЃерЃБрЃљрЃџрЃЮ
рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћ рЃгрЃарЃўрЃА рЃцрЃљрЃарЃЌрЃЮрЃЉрЃў: S = ¤ђ * r^2, рЃАрЃљрЃЊрЃљрЃф r рЃерЃћрЃДрЃЋрЃљрЃюрЃўрЃџрЃўрЃљ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃЏрЃўрЃћрЃа.

рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃўрЃЌ рЃарЃўрЃфрЃ«рЃЋрЃА 1-рЃЊрЃљрЃю 100-рЃЏрЃЊрЃћ.
