# ­ЪДх рЃЌрЃћрЃЏрЃљ: рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃћрЃЉрЃў Python-рЃерЃў

## ­Ъћ╣ рЃарЃљ рЃљрЃарЃўрЃА рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў (String)?
**рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў** Рђћ рЃћрЃА рЃљрЃарЃўрЃА рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃћрЃЉрЃўрЃА рЃЏрЃўрЃЏрЃЊрЃћрЃЋрЃарЃЮрЃЉрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃАрЃљрЃф рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ рЃбрЃћрЃЦрЃАрЃбрЃБрЃарЃў рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃюрЃљрЃ«рЃљрЃЊ.  
Python-рЃерЃў рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃћрЃЉрЃў рЃўрЃгрЃћрЃарЃћрЃЉрЃљ рЃЉрЃарЃГрЃДрЃљрЃџрЃћрЃЉрЃерЃў:

```python
text1 = "Hello"
text2 = 'Python'
text3 = """рЃћрЃА рЃљрЃарЃўрЃА
рЃЏрЃарЃљрЃЋрЃљрЃџрЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃљрЃюрЃў рЃбрЃћрЃЦрЃАрЃбрЃў."""
```

рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў рЃЏрЃўрЃћрЃЎрЃБрЃЌрЃЋрЃюрЃћрЃЉрЃљ **рЃАрЃљрЃЉрЃљрЃќрЃўрЃАрЃЮ рЃбрЃўрЃърЃА** `str`.

## ­Ъћ╣ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃЌрЃљрЃю рЃЊрЃљрЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃБрЃџрЃў рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃў

| рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў | рЃерЃћрЃЊрЃћрЃњрЃў |
|-----------|-----------|--------|
| рЃерЃћрЃћрЃарЃЌрЃћрЃЉрЃљ (concatenation) | `"Hi" + "Python"` | `"HiPython"` |
| рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃљ | `"Hi" * 3` | `"HiHiHi"` |
| рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃќрЃћ рЃгрЃЋрЃЊрЃЮрЃЏрЃљ | `"Python"[0]` | `'P'` |
| рЃАрЃўрЃњрЃарЃФрЃћ | `len("Python")` | `6` |
| рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃЊрЃљрЃГрЃарЃљ (slice) | `"Python"[1:4]` | `"yth"` |

## ­Ъћ╣ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃћрЃЉрЃЌрЃљрЃю рЃАрЃљрЃЏрЃБрЃерЃљрЃЮ рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў

### ­ЪДЕ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃћрЃЉрЃў
```python
text = "Python"
print(text.upper())     # рЃЊрЃўрЃЊрЃў рЃљрЃАрЃЮрЃћрЃЉрЃўрЃЌ
print(text.lower())     # рЃърЃљрЃбрЃљрЃарЃљ рЃљрЃАрЃЮрЃћрЃЉрЃўрЃЌ
print(text.capitalize())# рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃљрЃАрЃЮ рЃЊрЃўрЃЊрЃў
print(text.title())     # рЃЌрЃўрЃЌрЃЮрЃћрЃБрЃџрЃў рЃАрЃўрЃбрЃДрЃЋрЃўрЃА рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃљрЃАрЃЮ рЃЊрЃўрЃЊрЃў
print(text.swapcase())  # рЃЊрЃўрЃЊрЃў рЃљрЃАрЃЮ Рєњ рЃърЃљрЃбрЃљрЃарЃљ, рЃърЃљрЃбрЃљрЃарЃљ Рєњ рЃЊрЃўрЃЊрЃў
```

### ­ЪДЕ рЃњрЃљрЃЏрЃЮрЃбрЃЮрЃЋрЃћрЃЉрЃБрЃџрЃў рЃАрЃўрЃЋрЃарЃфрЃћрЃћрЃЉрЃў
```python
text = "   Hello World!   "
print(text.strip())   # рЃљрЃерЃЮрЃарЃћрЃЉрЃА рЃфрЃљрЃарЃўрЃћрЃџ рЃљрЃЊрЃњрЃўрЃџрЃћрЃЉрЃА рЃЮрЃарЃўрЃЋрЃћ рЃЏрЃ«рЃарЃўрЃЊрЃљрЃю
print(text.lstrip())  # рЃЏрЃљрЃарЃфрЃ«рЃюрЃўрЃЊрЃљрЃю
print(text.rstrip())  # рЃЏрЃљрЃарЃ»рЃЋрЃюрЃўрЃЊрЃљрЃю
```

### ­ЪДЕ рЃФрЃўрЃћрЃЉрЃљ рЃЊрЃљ рЃЕрЃљрЃюрЃљрЃфрЃЋрЃџрЃћрЃЉрЃљ
```python
text = "I love Python"
print(text.find("love"))     # рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃўрЃюрЃЊрЃћрЃЦрЃАрЃА 2
print(text.replace("Python", "GIS"))  # "I love GIS"
print("Python" in text)      # True
print("Java" not in text)    # True
```

## ­Ъћ╣ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃарЃћрЃЉрЃљ (String Formatting)

Python-рЃерЃў рЃњрЃЋрЃљрЃЦрЃЋрЃА рЃарЃљрЃЏрЃЊрЃћрЃюрЃўрЃЏрЃћ рЃњрЃќрЃљ рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃарЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА:

### 1№ИЈРЃБ f-string (рЃарЃћрЃЎрЃЮрЃЏрЃћрЃюрЃЊрЃћрЃЉрЃБрЃџрЃў)
```python
name = "Anna"
age = 21
print(f"My name is {name}, I am {age} years old.")
```

### 2№ИЈРЃБ `format()` рЃЏрЃћрЃЌрЃЮрЃЊрЃў
```python
text = "My name is {} and I am {} years old.".format("Anna", 21)
```

### 3№ИЈРЃБ %-рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃарЃћрЃЉрЃљ (рЃФрЃЋрЃћрЃџрЃў рЃАрЃбрЃўрЃџрЃў)
```python
name = "Anna"
age = 21
print("My name is %s and I am %d years old." % (name, age))
```

## ­Ъћ╣ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃћрЃЉрЃўрЃА рЃњрЃљрЃЊрЃљрЃЦрЃфрЃћрЃЋрЃљ рЃАрЃўрЃљрЃЊ рЃЊрЃљ рЃърЃўрЃарЃўрЃЦрЃўрЃЌ

```python
text = "apple,banana,orange"
fruits = text.split(",")   # ['apple', 'banana', 'orange']

joined = "-".join(fruits)  # 'apple-banana-orange'
```

## ­Ъћ╣ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮ рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃЎрЃџрЃћ рЃфрЃ«рЃарЃўрЃџрЃў

| рЃЏрЃћрЃЌрЃЮрЃЊрЃў | рЃљрЃдрЃгрЃћрЃарЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў |
|---------|---------|-----------|
| `.upper()` | рЃЊрЃўрЃЊрЃў рЃљрЃАрЃЮрЃћрЃЉрЃљрЃЊ рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃљ | `"hi".upper()` Рєњ `"HI"` |
| `.lower()` | рЃърЃљрЃбрЃљрЃарЃљ рЃљрЃАрЃЮрЃћрЃЉрЃљрЃЊ | `"HI".lower()" Рєњ `"hi"` |
| `.title()` | рЃАрЃљрЃЌрЃљрЃБрЃарЃўрЃА рЃцрЃЮрЃарЃЏрЃљ | `"hello world".title()` |
| `.strip()` | рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃљрЃЊрЃњрЃўрЃџрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃфрЃўрЃџрЃћрЃЉрЃљ | `"  hi ".strip()` |
| `.replace(a, b)` | рЃЕрЃљрЃюрЃљрЃфрЃЋрЃџрЃћрЃЉрЃљ | `"hi".replace("h","H")` |
| `.find(x)` | рЃърЃЮрЃЋрЃюрЃўрЃА рЃўрЃюрЃЊрЃћрЃЦрЃАрЃў | `"Python".find("th")` |
| `.count(x)` | рЃарЃљрЃЏрЃЊрЃћрЃюрЃ»рЃћрЃа рЃњрЃЋрЃ«рЃЋрЃЊрЃћрЃЉрЃљ x | `"banana".count("a")` |
| `.startswith(x)` | рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЌрЃБ рЃљрЃарЃљ | `"hello".startswith("he")` |
| `.endswith(x)` | рЃАрЃарЃБрЃџрЃЊрЃћрЃЉрЃљ рЃЌрЃБ рЃљрЃарЃљ | `"file.txt".endswith(".txt")` |

## ­ЪДа рЃАрЃљрЃЋрЃљрЃарЃ»рЃўрЃерЃЮрЃћрЃЉрЃў (20 рЃљрЃЏрЃЮрЃфрЃљрЃюрЃљ)

### рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃЊрЃЮрЃюрЃћ (1РђЊ10)
1. рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃћ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў рЃЊрЃљ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃЏрЃўрЃАрЃў рЃАрЃўрЃњрЃарЃФрЃћ.  
2. рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃЊрЃљ рЃЉрЃЮрЃџрЃЮ рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮ.  
3. рЃњрЃљрЃЊрЃљрЃљрЃЦрЃфрЃўрЃћ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃерЃћрЃДрЃЋрЃљрЃюрЃўрЃџрЃў рЃбрЃћрЃЦрЃАрЃбрЃў рЃЊрЃўрЃЊ рЃљрЃАрЃЮрЃћрЃЉрЃљрЃЊ.  
4. рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃЏрЃљ рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃЮрЃА рЃЮрЃарЃў рЃАрЃўрЃбрЃДрЃЋрЃљ Рђћ рЃњрЃљрЃљрЃћрЃарЃЌрЃўрЃљрЃюрЃћ рЃћрЃарЃЌ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃљрЃЊ.  
5. рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ рЃерЃћрЃўрЃфрЃљрЃЋрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў рЃАрЃўрЃбрЃДрЃЋрЃљрЃА `"Python"`.  
6. рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃЏрЃљ рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃЮрЃА рЃбрЃћрЃЦрЃАрЃбрЃў рЃЊрЃљ рЃЊрЃљрЃўрЃЌрЃЋрЃљрЃџрЃћ рЃарЃљрЃЏрЃЊрЃћрЃюрЃў `"a"` рЃљрЃарЃўрЃА рЃЏрЃљрЃАрЃерЃў.  
7. рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃерЃБрЃљ рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮ.  
8. рЃгрЃљрЃерЃљрЃџрЃћ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃгрЃўрЃю рЃЊрЃљ рЃБрЃЎрЃљрЃю рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃфрЃљрЃарЃўрЃћрЃџрЃў рЃљрЃЊрЃњрЃўрЃџрЃћрЃЉрЃў.  
9. рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ рЃўрЃгрЃДрЃћрЃЉрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў `"Hello"`-рЃЌрЃў.  
10. рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃўрЃА рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃћрЃЉрЃў рЃБрЃЎрЃБрЃдрЃЏрЃљ.

### рЃАрЃљрЃерЃБрЃљрЃџрЃЮ рЃАрЃўрЃарЃЌрЃБрЃџрЃћ (11РђЊ20)
11. рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃЏрЃљ рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃЮрЃА рЃгрЃўрЃюрЃљрЃЊрЃљрЃЊрЃћрЃЉрЃљ Рђћ рЃЊрЃљрЃўрЃЌрЃЋрЃљрЃџрЃћ рЃарЃљрЃЏрЃЊрЃћрЃюрЃў рЃАрЃўрЃбрЃДрЃЋрЃљрЃљ рЃЏрЃљрЃАрЃерЃў.  
12. рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃЏрЃљ рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃЮрЃА рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў Рђћ рЃерЃћрЃфрЃЋрЃљрЃџрЃћ рЃДрЃЋрЃћрЃџрЃљ `" "` `"_"`-рЃўрЃЌ.

## ­Ъћџ рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ

рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃћрЃЉрЃў Python-рЃерЃў рЃгрЃљрЃарЃЏрЃЮрЃљрЃЊрЃњрЃћрЃюрЃА рЃћрЃарЃЌ-рЃћрЃарЃЌ рЃДрЃЋрЃћрЃџрЃљрЃќрЃћ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃю рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃбрЃўрЃърЃА.  
рЃЏрЃљрЃЌрЃќрЃћ рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃњрЃљрЃюрЃЋрЃљрЃ«рЃЮрЃарЃфрЃўрЃћрЃџрЃЮрЃЌ:
- рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃА рЃгрЃЋрЃЊрЃЮрЃЏрЃљ рЃЊрЃљ рЃЊрЃљрЃГрЃарЃљ  
- рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃарЃћрЃЉрЃљ  
- рЃФрЃўрЃћрЃЉрЃљ рЃЊрЃљ рЃЕрЃљрЃюрЃљрЃфрЃЋрЃџрЃћрЃЉрЃљ  
- рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃўрЃА рЃцрЃљрЃарЃЌрЃЮ рЃАрЃърЃћрЃЦрЃбрЃарЃў рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃЏрЃЮрЃАрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃџрЃљрЃЊ  

­ЪЉЅ рЃЌрЃћрЃЏрЃўрЃА рЃЎрЃљрЃарЃњрЃљрЃЊ рЃЊрЃљрЃАрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃџрЃљрЃЊ рЃарЃћрЃЎрЃЮрЃЏрЃћрЃюрЃЊрЃћрЃЉрЃБрЃџрЃўрЃљ рЃљрЃЏрЃЮрЃфрЃљрЃюрЃћрЃЉрЃўрЃА рЃДрЃЋрЃћрЃџрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃўрЃА рЃњрЃљрЃерЃЋрЃћрЃЉрЃљ рЃЊрЃљ рЃбрЃћрЃАрЃбрЃўрЃарЃћрЃЉрЃљ.

def count_vowels(s):
  """рЃЊрЃљрЃўрЃЌрЃЋрЃџрЃўрЃА рЃўрЃюрЃњрЃџрЃўрЃАрЃБрЃарЃў рЃ«рЃЏрЃЮрЃЋрЃюрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃерЃў."""
  vowels = "aeiou"
  count = 0
  # рЃЋрЃљрЃЦрЃфрЃћрЃЋрЃЌ рЃърЃљрЃбрЃљрЃарЃљ рЃљрЃАрЃЮрЃћрЃЉрЃљрЃЊ, рЃарЃЮрЃЏ рЃЮрЃарЃўрЃЋрЃћ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ рЃЊрЃљрЃЋрЃўрЃЌрЃЋрЃљрЃџрЃЮрЃЌ
  for char in s.lower():
    if char in vowels:
      count += 1
  return count

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(count_vowels("Hello World!")) # 3 (e, o, o)

  """рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА рЃАрЃљрЃ«рЃћрЃџрЃўрЃА рЃўрЃюрЃўрЃфрЃўрЃљрЃџрЃћрЃЉрЃА."""
  words = full_name.split()
  initials = [word[0].upper() + '.' for word in words]
  return "".join(initials)

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(get_initials("рЃњрЃўрЃЮрЃарЃњрЃў рЃњрЃћрЃџрЃљрЃерЃЋрЃўрЃџрЃў")) # рЃњ.рЃњ.
print(get_initials("рЃ»рЃЮрЃю рЃАрЃЏрЃўрЃбрЃў")) # рЃ».рЃА.

def contains_only_digits(s):
  """рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА рЃерЃћрЃўрЃфрЃљрЃЋрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ рЃфрЃўрЃцрЃарЃћрЃЉрЃА."""
  return s.isdigit()

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(contains_only_digits("12345")) # True
print(contains_only_digits("123a45")) # False

# рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ (рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃАрЃљрЃ«рЃўрЃЌ)
def capitalize_every_other_char(s):
  """рЃДрЃЮрЃЋрЃћрЃџ рЃЏрЃћрЃЮрЃарЃћ рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃА рЃ«рЃЊрЃўрЃА рЃЊрЃўрЃЊ рЃљрЃАрЃЮрЃЊ (рЃўрЃюрЃЊрЃћрЃЦрЃАрЃўрЃЌ 1, 3, 5...)."""
  result = []
  for i, char in enumerate(s):
    if i % 2 != 0: # рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА рЃЎрЃћрЃюрЃбрЃў рЃўрЃюрЃЊрЃћрЃЦрЃАрЃўрЃА рЃЏрЃЦрЃЮрЃюрЃћ рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃћрЃЉрЃА
      result.append(char.upper())
    else:
      result.append(char)
  return "".join(result)

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(capitalize_every_other_char("abcdefg")) # aBcDeFg
print(capitalize_every_other_char("рЃњрЃљрЃЏрЃљрЃарЃ»рЃЮрЃЉрЃљ")) # рЃњр▓љрЃЏр▓љрЃар▓»рЃЮр▓ЉрЃљ

def count_words(s):
  """рЃЊрЃљрЃўрЃЌрЃЋрЃџрЃўрЃА рЃАрЃўрЃбрЃДрЃЋрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃерЃў."""
  words = s.split()
  return len(words)

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(count_words("рЃћрЃА рЃљрЃарЃўрЃА рЃАрЃљрЃЏрЃў рЃАрЃўрЃбрЃДрЃЋрЃљ")) # 4
print(count_words("рЃћрЃарЃЌрЃў")) # 1
print(count_words("  рЃАрЃцрЃћрЃўрЃАрЃћрЃЉрЃўрЃЌ  рЃЊрЃљрЃгрЃДрЃћрЃЉрЃБрЃџрЃў  ")) # 2

# рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ (рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃАрЃљрЃ«рЃўрЃЌ)
def format_phone_number(number_str):
  """рЃбрЃћрЃџрЃћрЃцрЃЮрЃюрЃўрЃА рЃюрЃЮрЃЏрЃћрЃарЃА рЃљрЃцрЃЮрЃарЃЏрЃўрЃарЃћрЃЉрЃА 'XXX-XX-XX-XX' рЃАрЃљрЃ«рЃћрЃерЃў (9-рЃюрЃўрЃерЃюрЃљ рЃюрЃЮрЃЏрЃарЃўрЃАрЃЌрЃЋрЃўрЃА)."""
  # рЃЋрЃерЃџрЃўрЃЌ рЃДрЃЋрЃћрЃџрЃљ рЃљрЃарЃљрЃфрЃўрЃцрЃарЃБрЃџ рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃА (рЃЌрЃБ рЃљрЃарЃўрЃА)
  digits = "".join(filter(str.isdigit, number_str))
  
  if len(digits) == 9:
    return f"{digits[0:3]}-{digits[3:5]}-{digits[5:7]}-{digits[7:9]}"
  else:
    return "рЃљрЃарЃљрЃАрЃгрЃЮрЃарЃў рЃюрЃЮрЃЏрЃарЃўрЃА рЃАрЃўрЃњрЃарЃФрЃћ"

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(format_phone_number("555123456")) # 555-12-34-56
print(format_phone_number("599-123-456")) # 599-12-34-56

def are_anagrams(s1, s2):
  """рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА рЃљрЃарЃўрЃљрЃю рЃЌрЃБ рЃљрЃарЃљ рЃЮрЃарЃў рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃў рЃљрЃюрЃљрЃњрЃарЃљрЃЏрЃћрЃЉрЃў."""
  # рЃЋрЃерЃџрЃўрЃЌ рЃАрЃцрЃћрЃўрЃАрЃћрЃЉрЃА рЃЊрЃљ рЃЋрЃљрЃЦрЃфрЃћрЃЋрЃЌ рЃърЃљрЃбрЃљрЃарЃљ рЃљрЃАрЃЮрЃћрЃЉрЃљрЃЊ
  cleaned_s1 = s1.lower().replace(' ', '')
  cleaned_s2 = s2.lower().replace(' ', '')
  
  # рЃљрЃюрЃљрЃњрЃарЃљрЃЏрЃћрЃЉрЃў рЃљрЃарЃўрЃљрЃю, рЃЌрЃБ рЃЏрЃљрЃЌрЃў рЃЊрЃљрЃџрЃљрЃњрЃћрЃЉрЃБрЃџрЃў рЃАрЃўрЃЏрЃЉрЃЮрЃџрЃЮрЃћрЃЉрЃўрЃА рЃАрЃўрЃљ рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃўрЃљ
  return sorted(cleaned_s1) == sorted(cleaned_s2)

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
print(are_anagrams("listen", "silent")) # True
print(are_anagrams("рЃћрЃџрЃћрЃЦрЃбрЃарЃЮ", "рЃбрЃћрЃџрЃћрЃЎрЃЮрЃЏ")) # True
print(are_anagrams("Hello", "World")) # False
