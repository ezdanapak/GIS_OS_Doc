# ­ЪДа Python РђЊ рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃбрЃўрЃърЃў рЃЊрЃљ рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў

## ­Ъћ╣ рЃерЃћрЃАрЃљрЃЋрЃљрЃџрЃў
**рЃњрЃљрЃЊрЃљрЃгрЃДрЃЋрЃћрЃбрЃўрЃџрЃћрЃЉрЃўрЃА рЃЏрЃўрЃдрЃћрЃЉрЃљ**.  
рЃЌрЃБ рЃарЃљрЃўрЃЏрЃћ рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ Рђћ рЃЋрЃљрЃЎрЃћрЃЌрЃћрЃЉрЃЌ рЃћрЃарЃЌ рЃарЃљрЃдрЃљрЃфрЃљрЃА, рЃ«рЃЮрЃџрЃЮ рЃЌрЃБ рЃљрЃарЃљ Рђћ рЃАрЃ«рЃЋрЃљрЃА.  
рЃљрЃЏрЃўрЃАрЃљрЃЌрЃЋрЃўрЃА Python-рЃерЃў рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃА **рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃбрЃўрЃърЃў (Boolean)** рЃЊрЃљ **рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў**.

---

## ­ЪћИ Boolean (рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў) рЃбрЃўрЃърЃў

Python-рЃерЃў рЃЮрЃарЃў рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃА:

```python
True   # рЃЏрЃљрЃарЃЌрЃљрЃџрЃў
False  # рЃЏрЃфрЃЊрЃљрЃарЃў
```

> рЃћрЃА рЃљрЃа рЃљрЃарЃўрЃА рЃАрЃбрЃарЃўрЃюрЃњрЃў, рЃљрЃЏрЃўрЃбрЃЮрЃЏ рЃљрЃа рЃБрЃюрЃЊрЃљ рЃЕрЃљрЃАрЃЋрЃљрЃЌ рЃЉрЃарЃГрЃДрЃљрЃџрЃћрЃЉрЃерЃў.

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:
```python
x = True
y = False

print(x)       # True
print(type(x)) # <class 'bool'>
```

> Boolean рЃбрЃўрЃърЃў рЃ«рЃерЃўрЃарЃљрЃЊ рЃЕрЃюрЃЊрЃћрЃЉрЃљ рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃАрЃљрЃА рЃљрЃю рЃърЃўрЃарЃЮрЃЉрЃћрЃЉрЃерЃў.

---

## ­ЪћИ рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў (Comparison Operators)

| рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃў | рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў | рЃерЃћрЃЊрЃћрЃњрЃў |
|------------|--------------|-----------|----------|
| `==` | рЃбрЃЮрЃџрЃўрЃљ | `5 == 5` | True |
| `!=` | рЃљрЃа рЃљрЃарЃўрЃА рЃбрЃЮрЃџрЃў | `5 != 3` | True |
| `>` | рЃЏрЃћрЃбрЃўрЃљ | `7 > 4` | True |
| `<` | рЃюрЃљрЃЎрЃџрЃћрЃЉрЃўрЃљ | `3 < 8` | True |
| `>=` | рЃЏрЃћрЃбрЃўрЃљ рЃљрЃю рЃбрЃЮрЃџрЃўрЃљ | `5 >= 5` | True |
| `<=` | рЃюрЃљрЃЎрЃџрЃћрЃЉрЃўрЃљ рЃљрЃю рЃбрЃЮрЃџрЃўрЃљ | `4 <= 6` | True |

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:
```python
age = 18
print(age >= 18)   # True
print(age < 21)    # True
print(age == 20)   # False
```

---

## ­ЪћИ рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў (Logical Operators)

| рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃў | рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљ | рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў | рЃерЃћрЃЊрЃћрЃњрЃў |
|------------|-------------|-----------|----------|
| `and` | рЃЊрЃљ | `True and True` | True |
| `or` | рЃљрЃю | `True or False` | True |
| `not` | рЃљрЃа | `not True` | False |

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:
```python
a = 5
b = 10

print(a > 3 and b < 15)  # True  (рЃЮрЃарЃўрЃЋрЃћ рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ)
print(a > 3 or b > 20)   # True  (рЃћрЃарЃЌрЃў рЃЏрЃљрЃўрЃюрЃф рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ)
print(not(a == 5))       # False (рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ, рЃљрЃЏрЃўрЃбрЃЮрЃЏ not рЃњрЃљрЃарЃЊрЃљрЃЦрЃЏрЃюрЃўрЃА False-рЃљрЃЊ)
```

---

## ­ЪћИ рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў (Conditional Statements)

рЃърЃўрЃарЃЮрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ `if`, `elif`, `else` рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА:

```python
if рЃърЃўрЃарЃЮрЃЉрЃљ:
    # рЃћрЃА рЃЉрЃџрЃЮрЃЎрЃў рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃћрЃЉрЃљ рЃЌрЃБ рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ
elif рЃАрЃ«рЃЋрЃљ_рЃърЃўрЃарЃЮрЃЉрЃљ:
    # рЃћрЃА рЃЉрЃџрЃЮрЃЎрЃў рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃћрЃЉрЃљ рЃЌрЃБ рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃфрЃЊрЃљрЃарЃўрЃљ, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃћрЃА рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ
else:
    # рЃћрЃА рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃћрЃЉрЃљ рЃЌрЃБ рЃљрЃарЃфрЃћрЃарЃЌрЃў рЃърЃўрЃарЃЮрЃЉрЃљ рЃљрЃа рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃљ
```

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:
```python
temperature = 25

if temperature > 30:
    print("рЃфрЃ«рЃћрЃџрЃљ!")
elif temperature >= 20:
    print("рЃЌрЃЉрЃўрЃџрЃљ!")
else:
    print("рЃфрЃўрЃЋрЃљ!")
```

­ЪЊц рЃерЃћрЃЊрЃћрЃњрЃў:
```
рЃЌрЃЉрЃўрЃџрЃљ!
```

## ­ЪћИ рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў (Nested if)

рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ `if` рЃЉрЃџрЃЮрЃЎрЃў рЃЕрЃљрЃЋрЃгрЃћрЃарЃЮрЃЌ рЃАрЃ«рЃЋрЃљ `if` рЃЉрЃџрЃЮрЃЎрЃерЃў:

```python
age = 20
citizen = True

if age >= 18:
    if citizen:
        print("рЃерЃћрЃњрЃўрЃФрЃџрЃўрЃљ рЃљрЃарЃЕрЃћрЃЋрЃюрЃћрЃЉрЃерЃў рЃЏрЃЮрЃюрЃљрЃгрЃўрЃџрЃћрЃЉрЃљ.")
    else:
        print("рЃЏрЃЮрЃЦрЃљрЃџрЃљрЃЦрЃћрЃЮрЃЉрЃљ рЃАрЃљрЃГрЃўрЃарЃЮрЃљ.")
else:
    print("рЃљрЃа рЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃў рЃ«рЃљрЃа.")
```

­ЪЊц рЃерЃћрЃЊрЃћрЃњрЃў:
```
рЃерЃћрЃњрЃўрЃФрЃџрЃўрЃљ рЃљрЃарЃЕрЃћрЃЋрЃюрЃћрЃЉрЃерЃў рЃЏрЃЮрЃюрЃљрЃгрЃўрЃџрЃћрЃЉрЃљ.
```

---

## ­ЪћИ рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃў рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃўрЃЊрЃљрЃю

### 1№ИЈРЃБ рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃў/рЃБрЃљрЃарЃДрЃЮрЃцрЃўрЃЌрЃў рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ
```python
num = -5

if num > 0:
    print("рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃљ")
elif num == 0:
    print("рЃюрЃБрЃџрЃЮрЃЋрЃљрЃюрЃўрЃљ")
else:
    print("рЃБрЃљрЃарЃДрЃЮрЃцрЃўрЃЌрЃўрЃљ")
```

---

### 2№ИЈРЃБ рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃўрЃА рЃЦрЃБрЃџрЃўрЃА рЃерЃћрЃцрЃљрЃАрЃћрЃЉрЃљ
```python
score = 82

if score >= 90:
    print("рЃерЃћрЃАрЃљрЃюрЃўрЃерЃюрЃљрЃЋрЃў")
elif score >= 75:
    print("рЃЎрЃљрЃарЃњрЃў")
elif score >= 60:
    print("рЃАрЃљрЃерЃБрЃљрЃџрЃЮ")
else:
    print("рЃЊрЃљрЃБрЃЎрЃЏрЃљрЃДрЃЮрЃцрЃўрЃџрЃћрЃЉрЃћрЃџрЃў")
```

---

### 3№ИЈРЃБ even/odd (рЃџрЃБрЃгрЃў рЃљрЃю рЃЎрЃћрЃюрЃбрЃў)
```python
number = 7

if number % 2 == 0:
    print("рЃџрЃБрЃгрЃўрЃљ")
else:
    print("рЃЎрЃћрЃюрЃбрЃўрЃљ")
```

---

### 4№ИЈРЃБ and / or рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ
```python
age = 25
has_ticket = True

if age >= 18 and has_ticket:
    print("рЃерЃћрЃњрЃўрЃФрЃџрЃўрЃљ рЃцрЃўрЃџрЃЏрЃќрЃћ рЃерЃћрЃАрЃЋрЃџрЃљ")
else:
    print("рЃерЃћрЃАрЃЋрЃџрЃљ рЃљрЃЎрЃарЃФрЃљрЃџрЃБрЃџрЃўрЃљ")
```

---

### 5№ИЈРЃБ not рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃў
```python
is_raining = False

if not is_raining:
    print("рЃгрЃљрЃЋрЃўрЃЊрЃћрЃЌ рЃњрЃљрЃАрЃћрЃўрЃарЃюрЃћрЃЉрЃљрЃЊ!")
else:
    print("рЃАрЃ»рЃЮрЃЉрЃА рЃАрЃљрЃ«рЃџрЃерЃў рЃЊрЃљрЃЋрЃарЃЕрЃћрЃЌ.")
```

## ­Ъћ╣ Boolean рЃарЃЮрЃњрЃЮрЃарЃф рЃерЃћрЃЊрЃћрЃњрЃў рЃАрЃ«рЃЋрЃљ рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃўрЃЊрЃљрЃю

рЃЏрЃљрЃЌрЃћрЃЏрЃљрЃбрЃўрЃЎрЃБрЃарЃў рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃћрЃЉрЃў рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃћрЃю Boolean рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃљрЃА:
```python
print(10 > 5)    # True
print(10 < 5)    # False
```

рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃћрЃА рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў рЃерЃћрЃЋрЃўрЃюрЃљрЃ«рЃЮрЃЌ рЃфрЃЋрЃџрЃљрЃЊрЃерЃў:
```python
result = (15 % 2 == 0)
print(result)  # False
```

---

## ­ЪћИ рЃћрЃарЃЌрЃў рЃ«рЃљрЃќрЃўрЃЌ рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЕрЃљрЃюрЃљрЃгрЃћрЃарЃў (Conditional Expression / Ternary Operator)

рЃўрЃюрЃљрЃ«рЃљрЃЋрЃА рЃерЃћрЃЊрЃћрЃњрЃА рЃћрЃарЃЌ рЃ«рЃљрЃќрЃерЃў:
```python
age = 20
status = "рЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃў" if age >= 18 else "рЃљрЃарЃљрЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃў"
print(status)
```

­ЪЊц рЃерЃћрЃЊрЃћрЃњрЃў:
```
рЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃў
```

---

## РюЁ рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ

- **Boolean рЃбрЃўрЃърЃў** Рєњ `True` / `False`  
- **рЃерЃћрЃЊрЃљрЃарЃћрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў** Рєњ `==`, `!=`, `>`, `<`, `>=`, `<=`  
- **рЃџрЃЮрЃњрЃўрЃЎрЃБрЃарЃў рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў** Рєњ `and`, `or`, `not`  
- **рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃћрЃЉрЃў** Рєњ `if`, `elif`, `else`

18№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃгрЃћрЃџрЃў рЃюрЃљрЃЎрЃўрЃљрЃюрЃў (4-рЃќрЃћ рЃўрЃДрЃЮрЃцрЃљ рЃЊрЃљ 100-рЃќрЃћ рЃљрЃарЃљ, рЃљрЃю 400-рЃќрЃћ рЃўрЃДрЃЮрЃцрЃљ).  
19№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃЮрЃарЃў рЃбрЃћрЃЦрЃАрЃбрЃў рЃљрЃАрЃЮ-рЃљрЃАрЃЮ, рЃарЃћрЃњрЃўрЃАрЃбрЃарЃўрЃА рЃБрЃњрЃБрЃџрЃћрЃЉрЃћрЃџрЃДрЃЮрЃцрЃўрЃЌ (`.lower()`).  
20№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃарЃўрЃюрЃњрЃерЃў рЃАрЃўрЃбрЃДрЃЋрЃљ `"Python"`.

---

## ­Ъћ╣ III рЃЊрЃЮрЃюрЃћ РђЊ рЃърЃўрЃарЃЮрЃЉрЃўрЃЌрЃў рЃљрЃарЃЕрЃћрЃЋрЃљрЃюрЃў (if / elif / else)

21№ИЈРЃБ рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА:  
- `"рЃфрЃўрЃЋрЃљ"` рЃЌрЃБ рЃбрЃћрЃЏрЃърЃћрЃарЃљрЃбрЃБрЃарЃљ < 15  
- `"рЃЌрЃЉрЃўрЃџрЃљ"` рЃЌрЃБ 15РђЊ25 рЃерЃЮрЃарЃўрЃАрЃљрЃљ  
- `"рЃфрЃ«рЃћрЃџрЃљ"` рЃЌрЃБ >25  

22№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃЦрЃБрЃџрЃљ рЃАрЃљрЃЎрЃЏрЃљрЃарЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃфрЃЊрЃўрЃА рЃЕрЃљрЃАрЃљрЃЉрЃљрЃарЃћрЃЉрЃџрЃљрЃЊ (>= 60).  
23№ИЈРЃБ рЃЦрЃБрЃџрЃћрЃЉрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃерЃћрЃцрЃљрЃАрЃћрЃЉрЃљ:  
- >=90: "рЃерЃћрЃАрЃљрЃюрЃўрЃерЃюрЃљрЃЋрЃў"  
- 75РђЊ89: "рЃЎрЃљрЃарЃњрЃў"  
- 60РђЊ74: "рЃАрЃљрЃерЃБрЃљрЃџрЃЮ"  
- <60: "рЃЊрЃљрЃБрЃЎрЃЏрЃљрЃДрЃЮрЃцрЃўрЃџрЃћрЃЉрЃћрЃџрЃў"

24№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃљрЃАрЃљрЃЎрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ:  
- <13 Рєњ "рЃЉрЃљрЃЋрЃерЃЋрЃў"  
- 13РђЊ17 Рєњ "рЃЏрЃЮрЃќрЃљрЃарЃЊрЃў"  
- 18РђЊ64 Рєњ "рЃќрЃарЃЊрЃљрЃАрЃарЃБрЃџрЃў"  
- >=65 Рєњ "рЃърЃћрЃюрЃАрЃўрЃЮрЃюрЃћрЃарЃў"

25№ИЈРЃБ рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА, рЃарЃљрЃЏрЃЊрЃћрЃюрЃў рЃфрЃўрЃцрЃарЃўрЃљ рЃарЃўрЃфрЃ«рЃЋрЃерЃў (`len(str(num))`).

## ­ЪћИ IV рЃЊрЃЮрЃюрЃћ РђЊ рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃърЃўрЃарЃЮрЃЉрЃћрЃЉрЃў (Nested if)

26№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃА рЃњрЃљрЃЏрЃЮрЃфрЃЊрЃўрЃА рЃЕрЃљрЃЉрЃљрЃарЃћрЃЉрЃљ:  
- рЃЦрЃБрЃџрЃљ РЅЦ 60  
- рЃЊрЃљрЃАрЃгрЃарЃћрЃЉрЃљ РЅЦ 75%  
рЃЌрЃБ рЃЮрЃарЃўрЃЋрЃћ рЃърЃўрЃарЃЮрЃЉрЃљ рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃљ Рєњ `"рЃЕрЃљрЃљрЃЉрЃљрЃарЃљ"`.

27№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃгрЃћрЃарЃА рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃАрЃљрЃ«рЃћрЃџрЃА рЃЊрЃљ рЃърЃљрЃарЃЮрЃџрЃА.  
рЃЌрЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃАрЃљрЃ«рЃћрЃџрЃў `"admin"` рЃЊрЃљ рЃърЃљрЃарЃЮрЃџрЃў `"1234"` Рєњ `"Welcome!"`,  
рЃАрЃ«рЃЋрЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃў `"Access denied"`.

28№ИЈРЃБ рЃЌрЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃерЃћрЃДрЃљрЃЋрЃА рЃљрЃАрЃљрЃЎрЃў рЃЊрЃљ рЃЏрЃЮрЃЦрЃљрЃџрЃљрЃЦрЃћрЃЮрЃЉрЃўрЃА рЃАрЃбрЃљрЃбрЃБрЃАрЃў (True/False),  
рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃљрЃарЃЕрЃћрЃЋрЃюрЃћрЃЉрЃерЃў рЃЏрЃЮрЃюрЃљрЃгрЃўрЃџрЃћрЃЮрЃЉрЃљ.

29№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃарЃўрЃфрЃ«рЃЋрЃў рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃў рЃЊрЃљ рЃљрЃАрЃћрЃЋрЃћ рЃџрЃБрЃгрЃў:  
рЃЌрЃБ рЃЎрЃў Рєњ `"рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃў рЃџрЃБрЃгрЃў"`,  
рЃЌрЃБ рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃўрЃљ, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃЎрЃћрЃюрЃбрЃў Рєњ `"рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃў рЃЎрЃћрЃюрЃбрЃў"`,  
рЃЌрЃБ рЃБрЃљрЃарЃДрЃЮрЃцрЃўрЃЌрЃўрЃљ Рєњ `"рЃБрЃљрЃарЃДрЃЮрЃцрЃўрЃЌрЃў"`.

30№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃљрЃАрЃљрЃЎрЃўрЃА рЃЊрЃљ рЃЉрЃўрЃџрЃћрЃЌрЃўрЃА рЃЦрЃЮрЃюрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ:  
- рЃљрЃЦрЃЋрЃА рЃЉрЃўрЃџрЃћрЃЌрЃў рЃЊрЃљ рЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃўрЃљ Рєњ `"рЃерЃћрЃњрЃўрЃФрЃџрЃўрЃљ рЃерЃћрЃАрЃЋрЃџрЃљ"`  
- рЃЉрЃўрЃџрЃћрЃЌрЃў рЃљрЃа рЃљрЃЦрЃЋрЃА Рєњ `"рЃерЃћрЃўрЃФрЃўрЃюрЃћ рЃЉрЃўрЃџрЃћрЃЌрЃў"`  
- рЃљрЃарЃљрЃАрЃарЃБрЃџрЃгрЃџрЃЮрЃЋрЃљрЃюрЃўрЃљ Рєњ `"рЃерЃћрЃАрЃЋрЃџрЃљ рЃљрЃЎрЃарЃФрЃљрЃџрЃБрЃџрЃўрЃљ"`

---

## ­Ъћ╣ V рЃЊрЃЮрЃюрЃћ РђЊ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃАрЃфрЃћрЃюрЃљрЃарЃћрЃЉрЃў

31№ИЈРЃБ рЃЏрЃљрЃдрЃљрЃќрЃўрЃљрЃерЃў рЃцрЃљрЃАрЃЊрЃљрЃЎрЃџрЃћрЃЉрЃљ:  
- рЃЌрЃБ рЃЌрЃљрЃюрЃ«рЃљ > 100 Рєњ 10% рЃцрЃљрЃАрЃЊрЃљрЃЎрЃџрЃћрЃЉрЃљ  
- рЃЌрЃБ рЃЌрЃљрЃюрЃ«рЃљ > 50 Рєњ 5% рЃцрЃљрЃАрЃЊрЃљрЃЎрЃџрЃћрЃЉрЃљ  
- рЃАрЃ«рЃЋрЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃў рЃцрЃљрЃАрЃЊрЃљрЃЎрЃџрЃћрЃЉрЃљ рЃљрЃарЃљрЃљ.

32№ИЈРЃБ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА рЃърЃљрЃарЃЮрЃџрЃўрЃА рЃАрЃўрЃФрЃџрЃўрЃћрЃарЃћрЃА:  
- рЃАрЃўрЃњрЃарЃФрЃћ >= 8  
- рЃерЃћрЃўрЃфрЃљрЃЋрЃА `"@"` рЃљрЃю `"!"`  
рЃЌрЃБ рЃЮрЃарЃўрЃЋрЃћ рЃърЃўрЃарЃЮрЃЉрЃљ рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃљ Рєњ `"рЃФрЃџрЃўрЃћрЃарЃў рЃърЃљрЃарЃЮрЃџрЃў"`

33№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃА рЃАрЃљрЃљрЃЌрЃА (0РђЊ23):  
рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃЊрЃдрЃћрЃўрЃА рЃЏрЃЮрЃюрЃљрЃЎрЃЋрЃћрЃЌрЃў:  
- 0РђЊ11 Рєњ `"рЃЊрЃўрЃџрЃљ"`  
- 12РђЊ17 Рєњ `"рЃерЃБрЃљрЃЊрЃдрЃћ"`  
- 18РђЊ23 Рєњ `"рЃАрЃљрЃдрЃљрЃЏрЃЮ"`

34№ИЈРЃБ рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃА, рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃърЃўрЃарЃўрЃА рЃерЃћрЃАрЃЋрЃџрЃљ рЃдрЃЮрЃюрЃўрЃАрЃФрЃўрЃћрЃЉрЃљрЃќрЃћ:  
- рЃљрЃАрЃљрЃЎрЃў >= 18  
- рЃЉрЃўрЃџрЃћрЃЌрЃў рЃљрЃЦрЃЋрЃА  
- рЃљрЃЊрЃњрЃўрЃџрЃў рЃЌрЃљрЃЋрЃўрЃАрЃБрЃцрЃљрЃџрЃўрЃљ  
рЃЌрЃБ рЃДрЃЋрЃћрЃџрЃљ рЃърЃўрЃарЃЮрЃЉрЃљ рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ Рєњ `"рЃерЃћрЃАрЃЋрЃџрЃљ рЃюрЃћрЃЉрЃљрЃЊрЃљрЃарЃЌрЃБрЃџрЃўрЃљ"`

35№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃўрЃбрЃДрЃЋрЃљ рЃърЃљрЃџрЃўрЃюрЃЊрЃарЃЮрЃЏрЃў (рЃгрЃўрЃю рЃЊрЃљ рЃБрЃЎрЃљрЃю рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃљрЃЊ рЃўрЃЎрЃўрЃЌрЃ«рЃћрЃЉрЃљ).

## ­ЪћИ VI рЃЊрЃЮрЃюрЃћ РђЊ рЃерЃћрЃЏрЃЮрЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃўрЃЌрЃў рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃћрЃЉрЃў

36№ИЈРЃБ рЃЊрЃљрЃгрЃћрЃарЃћ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃЏрЃўрЃћрЃа рЃерЃћрЃДрЃЋрЃљрЃюрЃўрЃџрЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ рЃљрЃЏрЃЉрЃЮрЃЉрЃА:
- `"Fizz"` рЃЌрЃБ 3-рЃќрЃћ рЃўрЃДрЃЮрЃцрЃљ,
- `"Buzz"` рЃЌрЃБ 5-рЃќрЃћ рЃўрЃДрЃЮрЃцрЃљ,
- `"FizzBuzz"` рЃЌрЃБ рЃЮрЃарЃўрЃЋрЃћрЃќрЃћ рЃўрЃДрЃЮрЃцрЃљ,
- рЃАрЃ«рЃЋрЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃў рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃЌрЃЋрЃўрЃЌрЃЮрЃю рЃарЃўрЃфрЃ«рЃЋрЃА.

37№ИЈРЃБ рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃЊрЃљрЃарЃћрЃЉрЃА рЃЮрЃа рЃарЃўрЃфрЃ«рЃЋрЃА рЃЊрЃљ рЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃарЃЮрЃЏрЃћрЃџрЃўрЃљ рЃЊрЃўрЃЊрЃў рЃљрЃю рЃбрЃЮрЃџрЃўрЃљ рЃЌрЃБ рЃљрЃарЃљ.  
38№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃгрЃћрЃарЃА рЃбрЃћрЃЏрЃърЃћрЃарЃљрЃбрЃБрЃарЃљрЃА рЃфрЃћрЃџрЃАрЃўрЃБрЃАрЃерЃў Рђћ рЃЌрЃБ >30 рЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА `"рЃФрЃљрЃџрЃўрЃљрЃю рЃфрЃ«рЃћрЃџрЃљ"`,  
рЃЌрЃБ 15РђЊ30 `"рЃюрЃЮрЃарЃЏрЃљрЃџрЃБрЃарЃў рЃбрЃћрЃЏрЃърЃћрЃарЃљрЃбрЃБрЃарЃљрЃљ"`,  
рЃЌрЃБ <15 `"рЃфрЃўрЃЋрЃљ"`.

39№ИЈРЃБ рЃерЃћрЃЦрЃЏрЃћрЃюрЃў рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃўрЃЌрЃЋрЃџрЃўрЃА, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃгрЃћрЃџрЃў "рЃюрЃљрЃЎрЃўрЃљрЃюрЃў" рЃЊрЃљ рЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃА рЃерЃћрЃбрЃДрЃЮрЃЉрЃўрЃюрЃћрЃЉрЃљрЃА.  
40№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃА рЃЦрЃБрЃџрЃљрЃА рЃЊрЃљ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ рЃЊрЃљрЃљрЃЊрЃњрЃћрЃюрЃА, рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃўрЃърЃћрЃюрЃЊрЃўрЃљрЃќрЃћ рЃњрЃљрЃюрЃљрЃфрЃ«рЃљрЃЊрЃўрЃА рЃњрЃљрЃЎрЃћрЃЌрЃћрЃЉрЃљ (>=85).

## ­Ъћ╣ VII рЃЊрЃЮрЃюрЃћ РђЊ Boolean рЃфрЃЋрЃџрЃљрЃЊрЃћрЃЉрЃўрЃЌ рЃЊрЃљ рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃћрЃЉрЃўрЃЌ

41№ИЈРЃБ `is_even = (number % 2 == 0)`  
рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃћ рЃбрЃћрЃЦрЃАрЃбрЃў `"рЃџрЃБрЃгрЃўрЃљ"` рЃЌрЃБ `is_even` рЃЏрЃљрЃарЃЌрЃљрЃџрЃўрЃљ, рЃАрЃ«рЃЋрЃљрЃњрЃЋрЃљрЃарЃљрЃЊ `"рЃЎрЃћрЃюрЃбрЃўрЃљ"`.

42№ИЈРЃБ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃерЃћрЃўрЃДрЃЋрЃљрЃюрЃА рЃЮрЃа рЃарЃўрЃфрЃ«рЃЋрЃА, рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃЮрЃарЃўрЃЋрЃћ рЃЊрЃљрЃЊрЃћрЃЉрЃўрЃЌрЃўрЃљ рЃЌрЃБ рЃљрЃарЃљ.  
43№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃбрЃћрЃЦрЃАрЃбрЃерЃў `"python"` рЃљрЃю `"java"`.  
44№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃа рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃбрЃћрЃЦрЃАрЃбрЃў рЃфрЃљрЃарЃўрЃћрЃџрЃў (`not text`).  
45№ИЈРЃБ рЃерЃћрЃљрЃЏрЃЮрЃгрЃЏрЃћ, рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃбрЃБрЃЊрЃћрЃюрЃбрЃў рЃњрЃљрЃЏрЃЮрЃфрЃЊрЃљрЃќрЃћ рЃЏрЃўрЃАрЃБрЃџрЃў рЃЊрЃљ рЃЕрЃљрЃЉрЃљрЃарЃћрЃЉрЃБрЃџрЃў (рЃЮрЃарЃў Boolean рЃфрЃЋрЃџрЃљрЃЊрЃў).
